```css
.reviews__carousel {
  /* Убран overflow: hidden */
  padding: 3rem 0 4rem; /* Увеличены отступы */
}

.reviews__carousel-wrapper {
  overflow: visible; /* Позволяет карточкам выходить за границы */
  padding: 20px 0; /* Вертикальный паддинг для hover */
}

.review-card:hover {
  transform: translateY(-10px); /* Только подъем, без масштаба */
  z-index: 100; /* Высокий z-index */
}
```
