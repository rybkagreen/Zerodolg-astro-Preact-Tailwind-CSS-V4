## Обновление конфигураций

### 📁 astro.config.mjs - Полностью переписан

```javascript
// Современные настройки
export default defineConfig({
  site: 'https://zerodolg.ru',
  output: 'static', // Явное указание

  build: {
    inlineStylesheets: 'auto', // Умная оптимизация
  },

  integrations: [
    preact({
      devtools: process.env.NODE_ENV === 'development', // DevTools
    }),
    sitemap({
      filter: (page) => !page.includes('/admin') && !page.includes('/api'), // Фильтрация
    }),
    robotsTxt({
      sitemap: 'https://zerodolg.ru/sitemap-index.xml', // Прямая ссылка
    }),
  ],

  vite: {
    build: {
      cssCodeSplit: true, // Tree-shaking для CSS
      rollupOptions: {
        output: {
          manualChunks: {
            preact: ['preact', '@preact/signals'], // Vendor splitting
          },
        },
      },
    },
  },
});
```
